def deactivate

def update
  @endpoint = Endpoint.find(params[:id])
  @endpoint.lat = params[:lat]
  @endpoint.lon = params[:lon]
  @endpoint.status = Endpoint.ACTIVE
  @endpoint.save
  respond_to { |format|
    format.json
  }
end
